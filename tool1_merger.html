function performMerge(data) { const dataBIndexMap = new Map(); const mergedList = []; // Build map for first array (dataA) data[0].forEach((item, index) => { dataBIndexMap.set(item.key, index); }); // Iterate through the second array (dataB) data[1].forEach(itemB => { const indexInA = dataBIndexMap.get(itemB.key); if (indexInA !== undefined) { // If key exists in dataA, merge mergedList.push({...itemB, ...data[0][indexInA]}); } else { // If key does not exist, just add itemB mergedList.push(itemB); } }); return mergedList; }
